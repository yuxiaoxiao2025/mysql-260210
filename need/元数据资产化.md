1.1 自动提取表血缘图谱
在你的项目中创建  schema_indexer.py ：
python
#!/usr/bin/env python3
# 用途：自动扫描900个表，构建轻量级知识图谱

import mysql.connector
import json
import hashlib
from sentence_transformers import SentenceTransformer
import chromadb

class SchemaGraphBuilder:
    def __init__(self, db_config):
        self.conn = mysql.connector.connect(**db_config)
        self.embedder = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')
        self.chroma_client = chromadb.PersistentClient(path="./chroma_db")
        self.collection = self.chroma_client.get_or_create_collection("table_metadata")
        
    def extract_all_tables(self):
        """提取所有表结构和关系"""
        cursor = self.conn.cursor(dictionary=True)
        
        # 获取所有表
        cursor.execute("""
            SELECT TABLE_NAME, TABLE_COMMENT 
            FROM information_schema.TABLES 
            WHERE TABLE_SCHEMA = DATABASE()
        """)
        tables = cursor.fetchall()
        
        metadata_list = []
        
        for table in tables:
            table_name = table['TABLE_NAME']
            comment = table['TABLE_COMMENT'] or ''
            
            # 获取字段信息
            cursor.execute(f"""
                SELECT COLUMN_NAME, DATA_TYPE, COLUMN_COMMENT, COLUMN_KEY
                FROM information_schema.COLUMNS 
                WHERE TABLE_SCHEMA = DATABASE() AND TABLE_NAME = '{table_name}'
            """)
            columns = cursor.fetchall()
            
            # 获取外键关系
            cursor.execute(f"""
                SELECT COLUMN_NAME, REFERENCED_TABLE_NAME, REFERENCED_COLUMN_NAME
                FROM information_schema.KEY_COLUMN_USAGE 
                WHERE TABLE_SCHEMA = DATABASE() 
                AND TABLE_NAME = '{table_name}' 
                AND REFERENCED_TABLE_NAME IS NOT NULL
            """)
            foreign_keys = cursor.fetchall()
            
            # 构建表描述文本（用于向量化）
            col_descriptions = [f"{c['COLUMN_NAME']}({c['DATA_TYPE']})" for c in columns]
            schema_text = f"表名：{table_name}。描述：{comment}。"
            schema_text += f"包含字段：{', '.join(col_descriptions)}。"
            schema_text += f"主业务：{self._infer_business_domain(table_name, comment)}"
            
            # 存储元数据
            metadata = {
                "table_name": table_name,
                "comment": comment,
                "columns": columns,
                "foreign_keys": foreign_keys,
                "schema_text": schema_text,
                "business_tags": self._extract_tags(table_name, comment)
            }
            metadata_list.append(metadata)
            
            # 存入向量数据库（用于语义检索）
            embedding = self.embedder.encode(schema_text).tolist()
            self.collection.add(
                ids=[table_name],
                embeddings=[embedding],
                metadatas=[{
                    "table_name": table_name,
                    "comment": comment,
                    "schema_summary": schema_text[:500]  # 截断存储
                }]
            )
            
        # 保存关系图谱JSON（轻量级，无需Neo4j）
        with open('table_graph.json', 'w', encoding='utf-8') as f:
            json.dump(metadata_list, f, ensure_ascii=False, indent=2)
            
        print(f"✅ 已索引 {len(metadata_list)} 个表，关系图谱已生成")
        
    def _infer_business_domain(self, table_name, comment):
        """推断业务领域（简单规则，可扩展）"""
        keywords = {
            "订单": ["order", "trade", "sale"],
            "用户": ["user", "customer", "member"],
            "库存": ["stock", "inventory", "warehouse"],
            "财务": ["finance", "account", "payment"]
        }
        text = (table_name + " " + comment).lower()
        for domain, keys in keywords.items():
            if any(k in text for k in keys):
                return domain
        return "其他"
        
    def _extract_tags(self, table_name, comment):
        """提取业务标签"""
        # 这里可以接入LLM做更智能的标签提取
        return [self._infer_business_domain(table_name, comment)]

if __name__ == "__main__":
    db_config = {
        "host": "localhost",
        "user": "root",
        "password": "your_password",
        "database": "your_db"
    }
    builder = SchemaGraphBuilder(db_config)
    builder.extract_all_tables()
